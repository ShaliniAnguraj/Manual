public class loops {
	
	public void whileloop() {
		//while loop --> initialization --> while(condition) -->
		//{ body of loop and increment
		 int a= 20;
		 while(a<=30) {
			 System.out.println(a);
			 a++;
			 }
	}
	
	public void dowhileloop() {
		// exit level   dowhile --> initialization --> do{ body increment/decrement }
		//while(condition)
		
		int m = 100;
		do {
			System.out.println(m);
			m--;
		}
		while(m>=1);
	}
	public void forloop() {
		//syntax(for initialization, condition , increment/decrement);
		
		for(int i = 1; i<=20; i++) {

		}
		
		
	}

public class Conditions {
	
	int a = 60, b= 20;
	
	public void ifcon() {
		if(a<b)
		{
		System.out.println(a);
		}
	}
	
	public void ifelsecon() {
		//if condition 
		//syntax: if(condition) --> true executed the body{ }
		if(a<b)
		{
		System.out.println(a);
		}
		else {
			System.out.println("invalid");
		}
	}
	public void elseifcon() {
		//elseif
		if(a<b)
		{
			System.out.println("greater");
		}
		else if(a==b)
		{
			System.out.println("less");
		}
		else if(a>=b)
		{
			System.out.println("equal");
		}
		else
		{
			System.out.println("invaild");
		}
	}
	
	public void switchcon() {
		//switch condition 
		//syntax switch(condition) switch to condition and print the body then break the case
		char  letter = 'u';
		
		switch (letter) {
		case'a':{
			System.out.println(a+b);
			break;
		}
		case'e':{
			System.out.println(a-b);
			break;
		}
		case'i':{
			System.out.println(a*b);
			break;
		}
		case'o':{
			System.out.println(a/b);
			break;
		}
		case'u':{
			System.out.println(a%b);
			break;
		}
		default:{
			System.out.println("invalid");
		}
		}
		
	}
	
	
	public static void main(String[] args) {
		Conditions ab = new Conditions();
		//ab.ifcon();
		//ab.ifelsecon();
		//ab.elseifcon();
		ab.switchcon();
		
	}

}

public class stringpalindrome {
	
	public static void main(String[]args) {
		int r,temp,sum=0;
		int n= 545;
		temp=n;
		
		while(n>0)
			
			r=n%10;
			sum=(sum*10)+r;
			n=n/10;
			
		if(temp==sum)
			System.out.println("Given string is palindrome ");
		else
			System.out.println("Given string is not a palindrome ");
	}

}
